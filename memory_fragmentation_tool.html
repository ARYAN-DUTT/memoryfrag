/* memfrag_auth.c — Authentication core (pure C) */
#include <stdio.h>
#include <string.h>
#include <stdint.h>
#include <stdbool.h>

#define MAX_USERS    32
#define HASH_LEN     64
#define SESSION_LEN  32

typedef struct {
    char     username[64];
    uint8_t  pwd_hash[HASH_LEN];
    uint32_t role;
    bool     active;
} User;

typedef struct {
    char     token[SESSION_LEN];
    uint32_t uid;
    time_t   expires;
} Session;

static User    user_db[MAX_USERS];
static Session sessions[MAX_USERS];

/* FNV-1a hash for password (simplified) */
static void hash_password(const char* pwd, uint8_t* out) {
    uint64_t h = 14695981039346656037ULL;
    while (*pwd) {
        h ^= (uint8_t)*pwd++;
        h *= 1099511628211ULL;
    }
    memcpy(out, &h, sizeof(h));
}

/* Constant-time comparison to prevent timing attacks */
static bool secure_compare(const uint8_t* a, const uint8_t* b, size_t n) {
    volatile uint8_t diff = 0;
    for (size_t i = 0; i < n; i++) diff |= a[i] ^ b[i];
    return diff == 0;
}

bool auth_login(const char* user, const char* pwd, char* token_out) {
    uint8_t h[HASH_LEN];
    hash_password(pwd, h);
    for (int i = 0; i < MAX_USERS; i++) {
        if (!user_db[i].active) continue;
        if (strcmp(user_db[i].username, user) != 0) continue;
        if (secure_compare(user_db[i].pwd_hash, h, HASH_LEN)) {
            generate_token(token_out); /* CSPRNG session token */
            return true;
        }
    }
    return false;
}

/* — Memory allocator (first/best/worst/next fit) — */
typedef struct Block {
    size_t        base, size, used;
    bool          free;
    char          pid[32];
    struct Block* next;
} Block;

Block* first_fit(Block* head, size_t sz) {
    for (Block* b = head; b; b = b->next)
        if (b->free && b->size >= sz) return b;
    return NULL;
}
